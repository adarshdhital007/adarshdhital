### Lexical Scope
```JS
// This is a simple JS code demonstrating identifiers, signs, strings, keywords, comments, JSX literals, properties, and entities

// Declare an identifier for a function
function greet(name) {
  // Using a string (which is also a value type)
  let message = `Hello, ${name}!`; // Template literal (JSX-like string)

  // Using a keyword `let` to declare a variable
  let entity = { 
    firstName: "John", // Property
    lastName: "Doe" // Property
  };

  // JS Sign (comparison sign)
  if (name === "Alice") {
    // JSX Literal in a return statement
    return <div>{message}</div>; // JSX-like string literal
  } else {
    // Single-line comment explaining this return
    return message; // Return message as string
  }
}

// Call the function with an identifier and string argument
console.log(greet("Alice")); // Call the function and output result

```
